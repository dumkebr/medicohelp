export const router = createBrowserRouter([
  { path: "/", element: <Navigate to="/login" replace /> },
  { path: "/login", element: <LoginPage /> },
  { path: "/register", element: <RegisterPage /> },
  { path: "/forgot", element: <ForgotPage /> },
  { path: "/verify-code", element: <VerifyCodePage /> },
  { path: "/profile", element: <Protected><ProfilePage /></Protected> },
  { path: "/atendimento", element: <Protected><AtendimentoPage /></Protected> },
  { path: "/pacientes", element: <Protected><PatientsListPage /></Protected> },
  { path: "/pacientes/novo", element: <Protected><NewPatientPage /></Protected> },
  { path: "/sobre", element: <AboutPage /> },
  { path: "*", element: <Navigate to="/login" replace /> },
]);

// Protected wrapper: se n√£o tiver token, redireciona para /login
function Protected({ children }) {
  const token = localStorage.getItem("token");
  if (!token) return <Navigate to="/login" replace />;
  return children;
}
