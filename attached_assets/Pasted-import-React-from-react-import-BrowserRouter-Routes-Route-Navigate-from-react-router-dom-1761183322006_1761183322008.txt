import ChatMedico from "./ChatMedico"; // seu componente principal do chat
import AdvancedHub from "./AdvancedHub"; // hub Avançado (calculadoras, posologia, pro)
import Calculators, { WellsCalc, GasometriaCalc, IGCalc } from "./Calculators"; // arquivo do canvas

function NotFoundRedirect() {
  return <Navigate to="/" replace />;
}

export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        {/* Principal */}
        <Route path="/" element={<ChatMedico />} />

        {/* Avançado (tela própria) */}
        <Route path="/avancado" element={<AdvancedHub />} />

        {/* Calculadoras */}
        <Route path="/calc" element={<Calculators />} />
        <Route path="/calc/wells" element={<WellsCalc />} />
        <Route path="/calc/gasometria" element={<GasometriaCalc />} />
        <Route path="/calc/ig" element={<IGCalc />} />

        {/* Fallback */}
        <Route path="*" element={<NotFoundRedirect />} />
      </Routes>
    </BrowserRouter>
  );
}
